50 EBPFRegisterPNA(const EBPF::EBPFProgram *program, cstring instanceName,
51 const IR::Declaration_Instance *di, EBPF::CodeGenInspector *codeGen)
52 : EBPF::EBPFTableBase(program, instanceName, codeGen) {
53 CHECK_NULL(di);
54 this->instanceName = di->toString();
55 if (!di->type->is<IR::Type_Specialized>()) {
56 ::error(ErrorType::ERR_MODEL, "Missing specialization: %1%", di);
57 return;
58 }
59 auto ts = di->type->to<IR::Type_Specialized>();
60
61 if (ts->arguments->size() != PARAM_INDEX_2) {
62 ::error(ErrorType::ERR_MODEL, "Expected a type specialized with two arguments: %1%",
63 ts);
64 return;
65 }
66
67 this->valueArg = ts->arguments->at(0);
68 this->keyArg = ts->arguments->at(1);
69
70 this->keyType = EBPF::EBPFTypeFactory::instance->create(keyArg);
71 this->valueType = EBPF::EBPFTypeFactory::instance->create(valueArg);
72 }
73 void emitRegisterRead(EBPF::CodeBuilder *builder, const P4::ExternMethod *method,
74 ControlBodyTranslatorPNA *translator,

82class EBPFTablePNADirectCounterPropertyVisitor : public EBPF::EBPFTablePsaPropertyVisitor {
83 public:
84 explicit EBPFTablePNADirectCounterPropertyVisitor(EBPF::EBPFTablePSA *table)
85 : EBPF::EBPFTablePsaPropertyVisitor(table) {}
86
87 bool preorder(const IR::PathExpression *pe) override {
88 auto decl = table->program->refMap->getDeclaration(pe->path, true);
89 auto di = decl->to<IR::Declaration_Instance>();
90 CHECK_NULL(di);
91 if (EBPF::EBPFObject::getSpecializedTypeName(di) != "DirectCounter") {
92 ::error(ErrorType::ERR_UNEXPECTED, "%1%: not a DirectCounter, see declaration of %2%",
93 pe, decl);
94 return false;
95 }
96 auto counterName = EBPF::EBPFObject::externalName(di);
97 auto tblname = table->table->container->name.originalName;
98 auto ctr = new EBPFCounterPNA(table->program, di, counterName, table->codeGen, tblname);
99 table->counters.emplace_back(std::make_pair(counterName, ctr));
100 return false;
101 }
102
103 void visitTableProperty() {
104 EBPF::EBPFTablePsaPropertyVisitor::visitTableProperty("pna_direct_counter"_cs);
105 }
106};
