380class MapIterChecked {
381 /* Iterate through a map (VECTOR(pair_t)), giving errors for non-string and
382 * duplicate keys (and skipping them) */
383 const VECTOR(pair_t) & map;
384 bool allow; // allow non-string keys
385 std::set<std::string> duplicates_allowed;
386 std::map<std::string, int> keys_seen;
387 class iter {
388 MapIterChecked *self;
389 pair_t *p;
390 void check() {
391 while (p != self->map.end()) {
392 if (self->allow && p->key.type != tSTR) break;
393 if (!CHECKTYPE(p->key, tSTR)) {
394 p++;
395 continue;
396 }
397 if (self->duplicates_allowed.count(p->key.s)) break;
398 if (self->keys_seen.count(p->key.s)) {
399 error(p->key.lineno, "Duplicate element %s", p->key.s);
400 warning(self->keys_seen[p->key.s], "previous element %s", p->key.s);
401 p++;
402 continue;
403 }
404 self->keys_seen[p->key.s] = p->key.lineno;
405 break;
406 }
407 }
408
409 public:
410 iter(MapIterChecked *s, pair_t *p_) : self(s), p(p_) { check(); }
411 pair_t &operator*() const { return *p; }
412 pair_t *operator->() const { return p; }
413 bool operator==(iter &a) const { return p == a.p; }
414 iter &operator++() {
415 p++;
416 check();
417 return *this;
418 }
419 };
420
421 public:
422 explicit MapIterChecked(const VECTOR(pair_t) & map_, bool o = false,
423 const std::set<std::string> &dup = {})
424 : map(map_), allow(o), duplicates_allowed(dup) {}
425 MapIterChecked(const VECTOR(pair_t) & map_, const std::set<std::string> &dup)
426 : map(map_), allow(false), duplicates_allowed(dup) {}
427 iter begin() { return iter(this, map.begin()); }
428 iter end() { return iter(this, map.end()); }
429};
