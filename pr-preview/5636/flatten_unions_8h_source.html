205class FlattenHeaderUnion : public PassManager {
206 public:
207 FlattenHeaderUnion(P4::ReferenceMap *refMap, P4::TypeMap *typeMap, bool loopsUnroll = true) {
208 passes.push_back(new P4::TypeChecking(refMap, typeMap));
209 // Lower push_front/pop_front on header union stacks first; the new nodes need
210 // types before HandleValidityHeaderUnion.
211 if (loopsUnroll) {
212 passes.push_back(new DoExpandHeaderUnionStackPushPop(typeMap));
213 passes.push_back(new P4::ClearTypeMap(typeMap));
214 passes.push_back(new P4::TypeInference(typeMap, false));
215 passes.push_back(new P4::TypeChecking(refMap, typeMap));
216 }
217 passes.push_back(new HandleValidityHeaderUnion(refMap, typeMap));
218 // Stack flattening is only applicable if parser loops are unrolled and
219 // header union stack elements are accessed using [] notation. This pass does not handle
220 // .next .last etc accessors for stack elements.
221 if (loopsUnroll) {
222 passes.push_back(new DoFlattenHeaderUnionStack(refMap, typeMap));
223 passes.push_back(new P4::RemoveAllUnusedDeclarations(RemoveUnusedPolicy()));
224 }
225 passes.push_back(new P4::ClearTypeMap(typeMap));
226 passes.push_back(new P4::TypeInference(typeMap, false));
227 passes.push_back(new P4::TypeChecking(refMap, typeMap));
228 passes.push_back(new DoFlattenHeaderUnion(refMap, typeMap));
229 passes.push_back(new P4::RemoveAllUnusedDeclarations(RemoveUnusedPolicy()));
230 passes.push_back(new P4::RemoveAllUnusedHUDDeclarations());
231 passes.push_back(new P4::ClearTypeMap(typeMap));
232 passes.push_back(new P4::TypeChecking(nullptr, typeMap));
233 passes.push_back(new P4::RemoveParserIfs(typeMap));
234 }
235};
