228 void generate(const IR::Node &v) {
229 out << "{" << std::endl;
230 ++indent;
231 if (node_refs.find(v.id) != node_refs.end()) {
232 out << indent << "\"Node_ID\" : " << v.id;
233 } else {
234 node_refs.insert(v.id);
235 v.toJSON(*this);
236 if (dumpSourceInfo) {
237 v.sourceInfoToJSON(*this);
238 }
239 }
240 out << std::endl << --indent << "}";
