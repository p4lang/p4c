25class CheckHeaderTypes : public Modifier {
26 const IR::V1Program *global = nullptr;
27
28 public:
29 CheckHeaderTypes() { setName("CheckHeaderTypes"); }
30 bool preorder(IR::V1Program *glob) override {
31 global = glob;
32 return true;
33 }
34 bool preorder(IR::Metadata *meta) override {
35 if (auto type = global->get<IR::v1HeaderType>(meta->type_name))
36 meta->type = type->as_metadata;
37 else
38 error(ErrorType::ERR_TYPE_ERROR, "%s: No header type %s", meta->srcInfo,
39 meta->type_name);
40 return true;
41 }
42 bool preorder(IR::HeaderOrMetadata *hdr) override {
43 if (auto type = global->get<IR::v1HeaderType>(hdr->type_name))
44 hdr->type = type->as_header;
45 else
46 error(ErrorType::ERR_TYPE_ERROR, "%s: No header type %s", hdr->srcInfo, hdr->type_name);
47 return true;
48 }
49};
