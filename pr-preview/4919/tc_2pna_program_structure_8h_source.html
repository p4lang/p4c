34class PnaProgramStructure : public P4::ProgramStructure {
35 protected:
36 P4::ReferenceMap *refMap;
37 P4::TypeMap *typeMap;
38
39 public:
40 // We place scalar user metadata fields (i.e., bit<>, bool)
41 // in the scalars map.
42 ordered_map<cstring, const IR::Declaration_Variable *> scalars;
43 unsigned scalars_width = 0;
44 unsigned error_width = 32;
45 unsigned bool_width = 1;
46
47 // architecture related information
48 ordered_map<const IR::Node *, block_t> block_type;
49
50 ordered_map<cstring, const IR::Type_Header *> header_types;
51 ordered_map<cstring, const IR::Type_Struct *> metadata_types;
52 ordered_map<cstring, const IR::Type_HeaderUnion *> header_union_types;
53 ordered_map<cstring, const IR::Declaration_Variable *> headers;
54 ordered_map<cstring, const IR::Declaration_Variable *> metadata;
55 ordered_map<cstring, const IR::Declaration_Variable *> header_stacks;
56 ordered_map<cstring, const IR::Declaration_Variable *> header_unions;
57 ordered_map<cstring, const IR::Type_Error *> errors;
58 ordered_map<cstring, const IR::Type_Enum *> enums;
59 ordered_map<cstring, const IR::P4Parser *> parsers;
60 ordered_map<cstring, const IR::P4ValueSet *> parse_vsets;
61 ordered_map<cstring, const IR::P4Control *> deparsers;
62 ordered_map<cstring, const IR::P4Control *> pipelines;
63 ordered_map<cstring, const IR::Declaration_Instance *> extern_instances;
64 ordered_map<cstring, cstring> field_aliases;
65
66 std::vector<const IR::ExternBlock *> globals;
67
68 public:
69 PnaProgramStructure(P4::ReferenceMap *refMap, P4::TypeMap *typeMap)
70 : refMap(refMap), typeMap(typeMap) {
71 CHECK_NULL(refMap);
72 CHECK_NULL(typeMap);
73 }
74
75 std::set<cstring> non_pipeline_controls;
76 std::set<cstring> pipeline_controls;
77
78 bool hasVisited(const IR::Type_StructLike *st) {
79 if (auto h = st->to<IR::Type_Header>())
80 return header_types.count(h->getName());
81 else if (auto s = st->to<IR::Type_Struct>())
82 return metadata_types.count(s->getName());
83 else if (auto u = st->to<IR::Type_HeaderUnion>())
84 return header_union_types.count(u->getName());
85 return false;
86 }
87
92 static bool isCounterMetadata(cstring ptName) { return !strcmp(ptName, "PNA_CounterType_t"); }
93
98 static bool isStandardMetadata(cstring ptName) {
99 return (!strcmp(ptName, "pna_main_parser_input_metadata_t") ||
100 !strcmp(ptName, "pna_main_input_metadata_t") ||
101 !strcmp(ptName, "pna_main_output_metadata_t"));
102 }
103};
