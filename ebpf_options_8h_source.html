22enum XDP2TC { XDP2TC_NONE, XDP2TC_META, XDP2TC_HEAD, XDP2TC_CPUMAP };

24class EbpfOptions : public CompilerOptions {
25 public:
27 std::filesystem::path outputFile;
29 bool loadIRFromJson = false;
31 bool emitExterns = false;
33 bool emitTraceMessages = false;
35 bool generateToXDP = false;
37 enum XDP2TC xdp2tcMode = XDP2TC_NONE;
39 unsigned int maxTernaryMasks = 128;
41 bool enableTableCache = false;
42
43 EbpfOptions();
44
45 void calculateXDP2TCMode() {
46 if (arch != "psa") {
47 return;
48 }
49
50 if (generateToXDP && xdp2tcMode == XDP2TC_META) {
51 std::cerr
52 << "XDP2TC 'meta' mode cannot be used if XDP is enabled. "
53 "Falling back to 'head' mode. For more information see "
54 "https://github.com/p4lang/p4c/blob/main/backends/ebpf/psa/README.md#xdp2tc-mode"
55 << std::endl;
56 xdp2tcMode = XDP2TC_HEAD;
57 } else if (generateToXDP && xdp2tcMode == XDP2TC_NONE) {
58 // use 'head' mode by default; it's the safest option.
59 std::cout << "Setting XDP2TC 'head' mode by default for XDP-based hook." << std::endl;
60 xdp2tcMode = XDP2TC_HEAD;
61 } else if (!generateToXDP && xdp2tcMode == XDP2TC_NONE) {
62 std::cout << "Setting XDP2TC 'meta' mode by default for TC-based hook." << std::endl;
63 // For TC, use 'meta' mode by default.
64 xdp2tcMode = XDP2TC_META;
65 }
66 BUG_CHECK(xdp2tcMode != XDP2TC_NONE, "xdp2tc mode should not be set to NONE, bug?");
67 }
68};
