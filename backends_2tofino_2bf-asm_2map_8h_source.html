224template <class M>
225class MapForKey {
226 M &map;
227 typename M::key_type key;
228 class iterator {
229 const MapForKey &self;
230 decltype(map.begin()) it;
231
232 public:
233 iterator(const MapForKey &s, decltype(map.begin()) i) : self(s), it(i) {}
234 iterator &operator++() {
235 if (++it != self.map.end() && it->first != self.key) it = self.map.end();
236 return *this;
237 }
238 bool operator==(const iterator &i) const { return it == i.it; }
239 bool operator!=(const iterator &i) const { return it != i.it; }
240 decltype(*&it->second) operator*() const { return it->second; }
241 decltype(&it->second) operator->() const { return &it->second; }
242 };
243
244 public:
245 MapForKey(M &m, typename M::key_type k) : map(m), key(k) {}
246 iterator begin() const { return iterator(*this, map.find(key)); }
247 iterator end() const { return iterator(*this, map.end()); }
248};
