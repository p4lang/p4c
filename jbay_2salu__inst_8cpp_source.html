93Instruction *MinMax::Decode::decode(Table *tbl, const Table::Actions::Action *act,
94 const VECTOR(value_t) & op) const {
95 auto *rv = new MinMax(this, op[0].lineno);
96 if (op.size > 2) {
97 if (op[1] == "phv")
98 rv->phv = true;
99 else if (op[1] != "mem")
100 error(op[1].lineno, "%s source must be 'mem' or 'phv'", op[0].s);
101 rv->mask = operand(tbl, act, op[2]);
102 if (!rv->mask.to<operand::Phv>() && !rv->mask.to<operand::Const>())
103 error(op[1].lineno, "%s mask must be constant or from phv or hash_dist", op[0].s);
104 } else {
105 error(op[0].lineno, "%s must have a single mask operand", op[0].s);
106 }
107 if (op.size == 4) {
108 rv->postmod = operand(tbl, act, op[3]);
109 } else if (op.size > 4) {
110 error(op[0].lineno, "too many operands for %s", op[0].s);
111 }
112 rv->slot = MINMAX;
113 return rv;
114}
115Instruction *MinMax::pass1(Table *tbl_, Table::Actions::Action *act) {
116 auto tbl = dynamic_cast<StatefulTable *>(tbl_);
117 BUG_CHECK(tbl);
118 int mask_size = (opc->opcode & 2) ? 8 : 16;
119 constval = boost::none;
120 mask->pass1(tbl);
121 act->minmax_use = true;
122 if (auto k = mask.to<operand::Const>()) {
123 if (k->value < 0 || k->value >= (1U << mask_size) || mask.neg)
124 error(k->lineno, "%s mask value out of range", name().c_str());
125 constval = k->value & ((1U << mask_size) - 1);
126 } else if (auto p = mask.to<operand::Phv>()) {
127 if (p->phv_index(tbl))
128 error(lineno, "%s phv mask must come from the lower half input", name().c_str());
129 } else {
130 error(mask->lineno, "%s invalid mask", name().c_str());
131 }
132 if (postmod) {
133 if (auto k = postmod.to<operand::Const>()) {
134 if (k->value < 0) {
135 k->value = -k->value;
136 postmod.neg = !postmod.neg;
137 }
138 if (k->value > 255) error(lineno, "%s post mod too large", name().c_str());
139 constval = constval.get_value_or(0) | (k->value & 0xff) << mask_size;
140 } else if (auto p = postmod.to<operand::Phv>()) {
141 if (!p->phv_index(tbl))
142 error(lineno, "%s phv post mod must come from the upper half input",
143 name().c_str());
144 } else {
145 error(postmod->lineno, "%s invalid post mod", name().c_str());
146 }
147 }
148 // We allocate the value here in order to report an error in pass1 if the capacity
149 // of the register file is exceeded. The next call in write_regs with the same value
150 // will return already allocated register file row index.
151 if (constval) tbl->get_const(lineno, *constval);
152 return this;
153}

159 auto tbl = dynamic_cast<StatefulTable *>(tbl_);
160 BUG_CHECK(tbl);
161 int logical_home_row = tbl->layout[0].row;
162 auto &meter_group = regs.rams.map_alu.meter_group[logical_home_row / 4U];
163 auto &salu_instr_common = meter_group.stateful.salu_instr_common[act->code];
164 if (auto k = mask.to<operand::Const>()) {
165 salu_instr_common.salu_minmax_mask_ctl = 1;
166 } else {
167 salu_instr_common.salu_minmax_mask_ctl = 0;
168 }
169 salu_instr_common.salu_minmax_ctl = opc->opcode;
170 salu_instr_common.salu_minmax_enable = 1;
171 if (postmod) {

173 salu_instr_common.salu_minmax_postmod_value_ctl = 0;
174 } else {
175 salu_instr_common.salu_minmax_postmod_value_ctl = 1;
176 }
177 if (postmod.neg)
178 salu_instr_common.salu_minmax_postdec_enable = 1;
179 else
180 salu_instr_common.salu_minmax_postinc_enable = 1;
181 }
182 if (constval) {
183 auto &salu_instr_cmp = meter_group.stateful.salu_instr_cmp_alu[act->code][3];
184 salu_instr_cmp.salu_cmp_regfile_adr = tbl->get_const(lineno, *constval);
185 }
186 // salu_instr_common.salu_minmax_src_sel = phv; -- FIXME -- specify PHV source?
187 for (auto &salu : meter_group.stateful.salu_instr_state_alu[act->code]) {
188 salu.salu_op = 0xd;
189 salu.salu_arith = 1;
190 salu.salu_pred = 0xffff;

195template <>
196void AluOP::write_regs(Target::JBay::mau_regs &regs, Table *tbl_, Table::Actions::Action *act) {
197 LOG2(this);
198 auto tbl = dynamic_cast<StatefulTable *>(tbl_);
199 BUG_CHECK(tbl);
200 int logical_home_row = tbl->layout[0].row;
201 auto &meter_group = regs.rams.map_alu.meter_group[logical_home_row / 4U];
202 auto &salu = meter_group.stateful.salu_instr_state_alu[act->code][slot - ALU2LO];
203 auto &salu_ext = meter_group.stateful.salu_instr2_state_alu[act->code][slot - ALU2LO];
204 auto &salu_instr_common = meter_group.stateful.salu_instr_common[act->code];
205 auto &salu_instr_output_alu = meter_group.stateful.salu_instr_output_alu[act->code];
206 salu.salu_op = opc->opcode & 0xf;
207 salu.salu_arith = opc->opcode >> 4;
208 salu.salu_pred = predication_encode;
209 bool need_flyover = (tbl->format->size >> tbl->is_dual_mode()) > 32;
210 const int alu_const_min = Target::STATEFUL_ALU_CONST_MIN();
211 const int alu_const_max = Target::STATEFUL_ALU_CONST_MAX();
212 if (srca) {
213 if (auto m = srca.to<operand::Memory>()) {
214 salu.salu_asrc_input = m->field->bit(0) > 0 ? 1 : 0;
215 if (need_flyover) {
216 salu_ext.salu_flyover_src_sel = 1;
217 need_flyover = false;
218 }
219 } else if (auto f = srca.to<operand::Phv>()) {
220 salu.salu_asrc_input = f->phv_index(tbl) ? 3 : 2;
221 if (need_flyover) {
222 salu_ext.salu_flyover_src_sel = 1;
223 need_flyover = false;
224 }
225 } else if (auto k = srca.to<operand::Const>()) {
226 salu.salu_asrc_input = 4;
227 if (k->value >= alu_const_min && k->value <= alu_const_max) {
228 salu.salu_const_src = k->value & Target::STATEFUL_ALU_CONST_MASK();
229 salu.salu_regfile_const = 0;
230 } else {
231 salu.salu_const_src = tbl->get_const(k->lineno, k->value);
232 salu.salu_regfile_const = 1;
233 }
234 } else if (auto r = srca.to<operand::Regfile>()) {
235 salu.salu_asrc_input = 4;
236 salu.salu_const_src = r->index;
237 salu.salu_regfile_const = 1;
238 } else {
239 BUG();
240 }
241 }
242 if (srcb) {
243 if (auto m = srcb.to<operand::Memory>()) {
244 salu.salu_bsrc_input = m->field->bit(0) > 0 ? 3 : 2;
245 if (need_flyover) {
246 salu_ext.salu_flyover_src_sel = 0;
247 need_flyover = false;
248 }
249 } else if (auto f = srcb.to<operand::Phv>()) {
250 salu.salu_bsrc_input = f->phv_index(tbl) ? 1 : 0;
251 if (need_flyover) {
252 salu_ext.salu_flyover_src_sel = 0;
253 need_flyover = false;
254 }
255 } else if (auto m = srcb.to<operand::MathFn>()) {
256 salu_instr_common.salu_alu2_lo_bsrc_math = 1;
257 if (auto b = m->of.to<operand::Phv>()) {
258 salu_instr_common.salu_alu2_lo_math_src = b->phv_index(tbl);
259 } else if (auto b = m->of.to<operand::Memory>()) {
260 salu_instr_common.salu_alu2_lo_math_src = b->field->bit(0) > 0 ? 3 : 2;
261 } else {
262 BUG();
263 }
264 } else if (auto k = srcb.to<operand::Const>()) {
265 salu.salu_bsrc_input = 4;
266 if (k->value >= alu_const_min && k->value <= alu_const_max) {
267 salu.salu_const_src = k->value & Target::STATEFUL_ALU_CONST_MASK();
268 salu.salu_regfile_const = 0;
269 } else {
270 salu.salu_const_src = tbl->get_const(k->lineno, k->value);
271 salu.salu_regfile_const = 1;
272 }
273 } else if (auto r = srcb.to<operand::Regfile>()) {
274 salu.salu_bsrc_input = 4;
275 salu.salu_const_src = r->index;
276 salu.salu_regfile_const = 1;
277 } else {
278 BUG();
279 }
280 }

301static int sbus_mask(int alu, const std::vector<Table::Ref> &tbls) {
302 int rv = 0;
303 for (auto &tbl : tbls) {
304 int bit = tbl->layout[0].row / 4U;
305 if (bit > alu) --bit;
306 rv |= 1 << bit;
307 }
308 return rv;

311template <>
312void CmpOP::write_regs(Target::JBay::mau_regs &regs, Table *tbl_, Table::Actions::Action *act) {
313 LOG2(this);
314 auto tbl = dynamic_cast<StatefulTable *>(tbl_);
315 BUG_CHECK(tbl);
316 int logical_home_row = tbl->layout[0].row;
317 auto &meter_group = regs.rams.map_alu.meter_group[logical_home_row / 4U];
318 auto &salu = meter_group.stateful.salu_instr_cmp_alu[act->code][slot];
319 auto &salu_instr_common = meter_group.stateful.salu_instr_common[act->code];
320 if (srca) {
321 salu.salu_cmp_asrc_input = srca->field->bit(0) > 0;
322 salu.salu_cmp_asrc_sign = srca_neg;
323 salu.salu_cmp_asrc_enable = 1;
324 if (maska != uint32_t(-1)) {
325 salu.salu_cmp_asrc_mask_enable = 1;
326 auto cval = 0;
327 if (auto k = dynamic_cast<const operand::Const *>(srcc))
328 cval = k->value;
329 else if (auto r = dynamic_cast<const operand::Regfile *>(srcc))
330 cval = tbl->get_const_val(r->index);
331 int64_t min = Target::STATEFUL_CMP_CONST_MIN();
332 int64_t max = Target::STATEFUL_CMP_CONST_MAX();
333 bool maska_outside = (maska < uint32_t(min) && maska > max);
334 bool maska_equal_inside = (uint32_t(cval) != maska && cval >= min && cval <= max);
335 if (!maska_outside && !maska_equal_inside) {
336 salu.salu_cmp_const_src = maska & Target::STATEFUL_CMP_CONST_MASK();
337 salu.salu_cmp_mask_input = 0;
338 } else {
339 salu.salu_cmp_regfile_adr = tbl->get_const(srca->lineno, maska);
340 salu.salu_cmp_mask_input = 1;
341 }
342 }
343 }
344 if (srcb) {
345 salu.salu_cmp_bsrc_input = srcb->phv_index(tbl);
346 salu.salu_cmp_bsrc_sign = srcb_neg;
347 salu.salu_cmp_bsrc_enable = 1;
348 if (maskb != uint32_t(-1)) {
349 // uarch 6.2.12.6.1, masks for operand a/b are sourced from the
350 // same regfile slot.
351 if (salu.salu_cmp_asrc_mask_enable && salu.salu_cmp_mask_input && maskb != maska)
352 error(lineno, "inconsistent operand mask %d and %d in salu compare operation",
353 maska, maskb);
354 salu.salu_cmp_bsrc_mask_enable = 1;
355 salu.salu_cmp_regfile_adr = tbl->get_const(srcb->lineno, maskb);
356 }
357 }
358 if (srcc) {
359 if (auto k = dynamic_cast<const operand::Const *>(srcc)) {
360 const int cmp_const_min = Target::STATEFUL_CMP_CONST_MIN();
361 const int cmp_const_max = Target::STATEFUL_CMP_CONST_MAX();
362 if (k->value >= cmp_const_min && k->value <= cmp_const_max) {
363 salu.salu_cmp_const_src = k->value & Target::STATEFUL_CMP_CONST_MASK();
364 salu.salu_cmp_regfile_const = 0;
365 } else {
366 // uarch 6.2.12.6.1, masks for operand a/b are sourced from the
367 // same regfile slot as operand c if c is a constant.
368 if (salu.salu_cmp_asrc_mask_enable && salu.salu_cmp_mask_input &&
369 maska != uint32_t(k->value))
370 error(lineno, "inconsistent operand mask %d and %d in salu compare operation",
371 maska, uint32_t(k->value));
372 if (salu.salu_cmp_bsrc_mask_enable && salu.salu_cmp_mask_input &&
373 maskb != uint32_t(k->value))
374 error(lineno, "inconsistent operand mask %d and %d in salu compare operation",
375 maskb, uint32_t(k->value));
376 salu.salu_cmp_regfile_adr = tbl->get_const(srcc->lineno, k->value);
377 salu.salu_cmp_regfile_const = 1;
378 }
379 } else if (auto r = dynamic_cast<const operand::Regfile *>(srcc)) {
380 salu.salu_cmp_regfile_adr = r->index;
381 salu.salu_cmp_regfile_const = 1;
382 }
383 } else {
384 salu.salu_cmp_const_src = 0;
385 salu.salu_cmp_regfile_const = 0;
386 }
387 salu.salu_cmp_opcode = opc->opcode | (type << 2);
388 auto lmask = sbus_mask(logical_home_row / 4U, tbl->sbus_learn);
389 auto mmask = sbus_mask(logical_home_row / 4U, tbl->sbus_match);
390 salu_instr_common.salu_lmatch_sbus_listen = lmask;
391 salu_instr_common.salu_match_sbus_listen = mmask;
392 salu_instr_common.salu_sbus_in_comb = tbl->sbus_comb;
393 if (lmask || mmask) {
394 // if lmask and mmask are both zero, these registers don't matter, but the model
395 // will assert if they are non-zero)
396 salu.salu_cmp_sbus_or = 0;
397 salu.salu_cmp_sbus_and = learn ? 1 : 0;
398 salu.salu_cmp_sbus_invert = learn_not ? 1 : 0;
399 }
400}

439void OutOP::decode_output_mux(Target::JBay, Table *tbl, value_t &op) {
440 static const std::map<std::string, int> ops_mux_lookup = {
441 {"mem_hi", 1}, {"mem_lo", 0}, {"memory_hi", 1}, {"memory_lo", 0}, {"phv_hi", 3},
442 {"phv_lo", 2}, {"alu_hi", 5}, {"alu_lo", 4}, {"minmax_index", 5}, {"minmax_post", 4},
443 {"predicate", 6}, {"address", 7}, {"div", 8}, {"mod", 9}, {"minmax", 10}};
444 if (op.type == tCMD && ops_mux_lookup.count(op[0].s))
445 output_mux = ops_mux_lookup.at(op[0].s);
446 else if (op.type == tSTR && ops_mux_lookup.count(op.s))
447 output_mux = ops_mux_lookup.at(op.s);
448 else
449 output_mux = -1;
450 if (src) {
451 int tmp = output_mux;
452 if (auto *phv = src.to<operand::Phv>())
453 output_mux = 2 + phv->phv_index(tbl->to<StatefulTable>());
454 else if (auto *mem = src.to<operand::Memory>())
455 output_mux = mem->field->bit(0) > 0 ? 1 : 0;
456 BUG_CHECK(tmp < 0 || tmp == output_mux, "inconsistent output mux decode");
457 }
458}
459int OutOP::decode_output_option(Target::JBay, value_t &op) {
460 if (op == "lmatch") {
461 lmatch = true;
462 if (op.type == tCMD)
463 lmatch_pred = decode_predicate(op[1]);
464 else
465 lmatch_pred = STATEFUL_PREDICATION_ENCODE_UNCOND;
466 } else {
467 return -1;
468 }
469 return 0;
470}
