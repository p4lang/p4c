37 // Clone expressions of the form Member(TypeNameExpression)
38 const IR::Node *postorder(IR::Member *member) override {
39 if (member->expr->is<IR::TypeNameExpression>()) {
40 return new IR::Member(member->expr->clone(), member->member);
41 }
42 return member;
43 }
