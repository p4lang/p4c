134/* disable power gating configuration for specific MAU regs to weedout delay programming
135 * issues. We dont expect to call this function in the normal usage of JBay - this is
136 * only for emulator debug/bringup
137 */
138template <class REGS>
139static void disable_jbay_power_gating(REGS &regs) {
140 for (gress_t gress : Range(INGRESS, EGRESS)) {
141 regs.dp.mau_match_input_xbar_exact_match_enable[gress] |= 0x1;
142 regs.dp.xbar_hash.xbar.mau_match_input_xbar_ternary_match_enable[gress] |= 0x1;
143 }
144
145 auto &xbar_power_ctl = regs.dp.match_input_xbar_din_power_ctl;
146 auto &actionmux_power_ctl = regs.dp.actionmux_din_power_ctl;
147 for (int side = 0; side < 2; side++) {
148 for (int reg = 0; reg < 16; reg++) {
149 xbar_power_ctl[side][reg] |= 0x3FF;
150 actionmux_power_ctl[side][reg] |= 0x3FF;
151 }
152 }

155template <>
156void Stage::write_regs(Target::JBay::mau_regs &regs, bool) {
157 write_common_regs<Target::JBay>(regs);
158 auto &merge = regs.rams.match.merge;
159 for (gress_t gress : Range(INGRESS, EGRESS)) {
160 if (stageno == 0) {
161 merge.predication_ctl[gress].start_table_fifo_delay0 = pred_cycle(gress) - 2;
162 merge.predication_ctl[gress].start_table_fifo_enable = 1;
163 } else if (stage_dep[gress] == MATCH_DEP) {
164 merge.predication_ctl[gress].start_table_fifo_delay0 =
165 this[-1].pipelength(gress) - this[-1].pred_cycle(gress) + pred_cycle(gress) - 3;
166 merge.predication_ctl[gress].start_table_fifo_enable = 1;
167 } else {
168 BUG_CHECK(stage_dep[gress] == ACTION_DEP);
169 merge.predication_ctl[gress].start_table_fifo_delay0 = 0;
170 merge.predication_ctl[gress].start_table_fifo_enable = 0;
171 }
172
173 if (stageno != 0)
174 regs.dp.cur_stage_dependency_on_prev[gress] = stage_dep[gress] != MATCH_DEP;
175
176 /* set stage0 dependency if explicitly set by the commandline option */
177 if (stageno == 0 && !options.stage_dependency_pattern.empty())
178 regs.dp.cur_stage_dependency_on_prev[gress] = stage_dep[gress] != MATCH_DEP;
179
180 if (stageno != AsmStage::numstages() - 1)
181 regs.dp.next_stage_dependency_on_cur[gress] = this[1].stage_dep[gress] != MATCH_DEP;
182 else if (AsmStage::numstages() < Target::NUM_MAU_STAGES())
183 regs.dp.next_stage_dependency_on_cur[gress] = 1;
184 auto &deferred_eop_bus_delay = regs.rams.match.adrdist.deferred_eop_bus_delay[gress];
185 deferred_eop_bus_delay.eop_internal_delay_fifo = pred_cycle(gress) + 2;
186 /* FIXME -- making this depend on the dependency of the next stage seems wrong */
187 if (stageno == AsmStage::numstages() - 1) {
188 if (AsmStage::numstages() < Target::NUM_MAU_STAGES())
189 deferred_eop_bus_delay.eop_output_delay_fifo = 1;
190 else
191 deferred_eop_bus_delay.eop_output_delay_fifo = pipelength(gress) - 2;
192 } else if (this[1].stage_dep[gress] == MATCH_DEP) {
193 deferred_eop_bus_delay.eop_output_delay_fifo = pipelength(gress) - 2;
194 } else {
195 deferred_eop_bus_delay.eop_output_delay_fifo = 1;
196 }
197 deferred_eop_bus_delay.eop_delay_fifo_en = 1;
198 if (stageno != AsmStage::numstages() - 1 && this[1].stage_dep[gress] == MATCH_DEP) {
199 merge.mpr_thread_delay[gress] = pipelength(gress) - pred_cycle(gress) - 4;
200 } else {
201 /* last stage in JBay must be always set as match-dependent on deparser */
202 if (stageno == AsmStage::numstages() - 1) {
203 merge.mpr_thread_delay[gress] = pipelength(gress) - pred_cycle(gress) - 4;
204 } else {
205 merge.mpr_thread_delay[gress] = 0;
206 }
207 }
208 }
209
210 for (gress_t gress : Range(INGRESS, EGRESS))
211 if (table_use[gress] & USE_TCAM)
212 regs.tcams.tcam_piped |= options.match_compiler ? 3 : 1 << gress;
213
214 for (gress_t gress : Range(INGRESS, EGRESS)) {
215 regs.cfg_regs.amod_pre_drain_delay[gress] = pipelength(gress) - 9;
216 if (this[1].stage_dep[gress] == MATCH_DEP)
217 regs.cfg_regs.amod_wide_bubble_rsp_delay[gress] = pipelength(gress) - 3;
218 else
219 regs.cfg_regs.amod_wide_bubble_rsp_delay[gress] = 0;
220 }
221 /* Max re-request limit with a long interval. Parb is going to have a large
222 * gap configured to minimize traffic hits during configuration this means
223 * that individual stages may not get their bubbles and will need to retry. */
224 regs.cfg_regs.amod_req_interval = 6732;
225 regs.cfg_regs.amod_req_limit = 15;
226
227 if (stageno == 0) {
228 /* MFerrera: "After some debug on the emulator, we've found a programming issue due to
229 * incorrect documentation and CSR description of match_ie_input_mux_sel in JBAY"
230 * MAU Stage 0 must always be configured to source iPHV from Parser-Arbiter
231 * Otherwise, MAU stage 0 is configured as match-dependent on Parser-Arbiter */
232 regs.dp.match_ie_input_mux_sel |= 3;
233 }
234
235 merge.pred_stage_id = stageno;
236 if (long_branch_terminate) merge.pred_long_brch_terminate = long_branch_terminate;
237 for (gress_t gress : Range(INGRESS, GHOST)) {
238 if (long_branch_thread[gress])
239 merge.pred_long_brch_thread[gress] = long_branch_thread[gress];
240 }
241
242 for (gress_t gress : Range(INGRESS, GHOST)) {
243 merge.mpr_stage_id[gress] = mpr_stage_id[gress];
244 for (int id = 0; id < LOGICAL_TABLES_PER_STAGE; ++id) {
245 merge.mpr_next_table_lut[gress][id] = mpr_next_table_lut[gress][id];
246 }
247 }
248 for (int id = 0; id < LOGICAL_TABLES_PER_STAGE; ++id) {
249 merge.mpr_glob_exec_lut[id] = mpr_glob_exec_lut[id];
250 }
251 for (int id = 0; id < MAX_LONGBRANCH_TAGS; ++id) {
252 merge.mpr_long_brch_lut[id] = mpr_long_brch_lut[id];
253 }
254 merge.mpr_always_run = mpr_always_run;
255
256 if (stageno != AsmStage::numstages() - 1) {
257 merge.mpr_bus_dep.mpr_bus_dep_ingress = this[1].stage_dep[INGRESS] != MATCH_DEP;
258 merge.mpr_bus_dep.mpr_bus_dep_egress = this[1].stage_dep[EGRESS] != MATCH_DEP;
259 }
260
261 merge.mpr_bus_dep.mpr_bus_dep_glob_exec = mpr_bus_dep_glob_exec[INGRESS] |
262 mpr_bus_dep_glob_exec[EGRESS] |
263 mpr_bus_dep_glob_exec[GHOST];
264 merge.mpr_bus_dep.mpr_bus_dep_long_brch = mpr_bus_dep_long_branch[INGRESS] |
265 mpr_bus_dep_long_branch[EGRESS] |
266 mpr_bus_dep_long_branch[GHOST];
267
268 merge.mpr_long_brch_thread = long_branch_thread[EGRESS];
269 if (auto conflict = (long_branch_thread[INGRESS] | long_branch_thread[GHOST]) &
270 long_branch_thread[EGRESS]) {
271 // Should probably check this earlier, but there's not a good place to do it.
272 for (auto tag : bitvec(conflict)) {
273 error(long_branch_use[tag]->lineno,
274 "Need one-stage turnaround before reusing "
275 "long_branch tag %d in a different thread",
276 tag);
277 }
278 }
279
280 bitvec in_use = match_use[INGRESS] | action_use[INGRESS] | action_set[INGRESS];
281 bitvec eg_use = match_use[EGRESS] | action_use[EGRESS] | action_set[EGRESS];
282 /* FIXME -- if the regs are live across a stage (even if not used in that stage) they
283 * need to be set in the thread registers. For now we just assume if they are used
284 * anywhere, they need to be marked as live */
285 in_use |= Phv::use(INGRESS);
286 eg_use |= Phv::use(EGRESS);
287 static const int phv_use_transpose[2][14] = {
288 {0, 1, 2, 3, 8, 9, 10, 11, 16, 17, 18, 19, 20, 21},
289 {4, 5, 6, 7, 12, 13, 14, 15, 22, 23, 24, 25, 26, 27}};
290 // FIXME -- this code depends on the Phv::Register uids matching the
291 // FIXME -- mau encoding of phv containers. (FIXME-PHV)
292 for (int i = 0; i < 2; i++) {
293 for (int j = 0; j < 14; j++) {
294 regs.dp.phv_ingress_thread[i][j] = regs.dp.phv_ingress_thread_imem[i][j] =
295 in_use.getrange(10 * phv_use_transpose[i][j], 10);
296 regs.dp.phv_egress_thread[i][j] = regs.dp.phv_egress_thread_imem[i][j] =
297 eg_use.getrange(10 * phv_use_transpose[i][j], 10);
298 }
299 }
300
301 /* Things following are for debug/bringup only : not for normal flows */
302
303 if (options.disable_power_gating) {
304 disable_jbay_power_gating(regs);
305 }
306
307 write_teop_regs(regs);
308}
