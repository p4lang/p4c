157class VersionIter {
158 unsigned left, bit;
159 void check() {
160 while (left && !(left & 1)) {
161 ++bit;
162 left >>= 1;
163 }
164 }
165 VersionIter() : left(0), bit(0) {}
166
167 public:
168 explicit VersionIter(config_version_t v) : left(v), bit(0) { check(); }
169 VersionIter begin() { return *this; }
170 VersionIter end() { return VersionIter(); }
171 int operator*() const { return bit; }
172 bool operator==(VersionIter &a) { return (left << bit) == (a.left << a.bit); }
173 VersionIter &operator++() {
174 left &= ~1;

176 return *this;
177 }
